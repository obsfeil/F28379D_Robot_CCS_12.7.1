486// SWI_isr, Using this interrupt as a Software started interrupt
487__interrupt void SWI_isr(void) {
488
489 // These three lines of code allow SWI_isr, to be interrupted by other interrupt functions
490 // making it lower priority than all other Hardware interrupts.
491 PieCtrlRegs.PIEACK.all = PIEACK_GROUP12;
492 asm(" NOP"); // Wait one cycle
493 EINT; // Clear INTM to enable interrupts
494
495
496
497 // Insert SWI ISR Code here.......
498 //solve for filtered velocity in rads/s
499 LeftVel = .6*LeftVel_1 + 100*(LeftWheel-LeftWheel_1);
500 RightVel = .6*RightVel_1 + 100*(RightWheel-RightWheel_1);
501
502 //Turning
503 WheelDif = LeftWheel - RightWheel;
504 velWheelDif = .33*velWheelDif_1 + .67*(WheelDif-WheelDif_1)/.004;
505 eDif = turnref - WheelDif;
506 //calculate iDif without integral windup
507 if(fabs(turn)>3){
508 iDif = iDif_1;
509 }else{
510 iDif = iDif_1 + .004*(eDif+eDif_1)/2;
511 }
512
513 //Calculate Balance and Turn COntrols
514 ubal = -K1*tilt_value -K2*gyro_value - K3*(LeftVel+RightVel)/2.0;
515 turn = Kp*eDif + Ki*iDif - Kd*velWheelDif;
516 //saturate turn so balance control dominates (controllefford between +-10)
517 if(turn>4){
518 turn = 4;
519 }else if(turn<-4){
520 turn = -4;
521 }
522
523 turnref = turnref_1 + (turnrate + turnrate_1)*.004/2.0;
524
525 //Calculate controleffort for each motor
526 uRight = ubal/2.0 - turn + FwdBackOffset;
527 uLeft = ubal/2.0 + turn + FwdBackOffset;
528
529 //Send ControlEffort to PWM Motor Command
530 setEPWM2A(uRight);
531 setEPWM2B(-uLeft);
532
533 //Save Past States
534 LeftVel_1 = LeftVel;
535 RightVel_1 = RightVel;
536 LeftWheel_1 = LeftWheel;
537 RightWheel_1 = RightWheel;
538 eDif_1 = eDif;
539 iDif_1 = iDif;
540 turnrate_1 = turnrate;
541 turnref_1 = turnref;
542
543 numSWIcalls++;
544
545 DINT;
546
547}
