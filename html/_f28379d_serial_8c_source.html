109
110 sci->SCICTL1.bit.TXWAKE = 0;

112 sci->SCICTL1.bit.RXENA = 1; // enable SCI receiver
113 sci->SCICTL1.bit.RXERRINTENA = 0; // disable receive error interrupt

137 PieCtrlRegs.PIEIER8.bit.INTx8 = 1;
138 PieCtrlRegs.PIEACK.all = (PIEACK_GROUP8);

142 return 0;

179/***************************************************************************
180 * SERIAL_SEND()
181 *
182 * "User level" function to send data via serial. Return value is the
183 * length of data successfully copied to the TX buffer.
184 ***************************************************************************/
185
186uint16_t serial_send(serial_t *s, char *data, uint16_t len)
187{
188 uint16_t i = 0;
189 if (len && s->TX.size < BUF_SIZE) {
190 for (i = 0; i < len; i++) {
191 if (buf_write_1(&s->TX, data[i] & 0x00FF) != 0) break;
192 }
193 s->sci->SCIFFTX.bit.TXFFINTCLR = 1; // TX: clear interrupt flag
194 s->sci->SCIFFTX.bit.TXFFIENA = 1; // TX: enable fifo interrupt
195 }
196 return i;
197}

293static inline void serial_recv_ready(serial_t *s)
294{
295 char data = s->sci->SCIRXBUF.all;
296
297 /* SCI PE or FE error */
298 if (data & 0xC000) {
299 s->sci->SCICTL1.bit.SWRESET = 0;
300 s->sci->SCICTL1.bit.SWRESET = 1;
301 s->sci->SCIFFRX.bit.RXFIFORESET = 0;
302 s->sci->SCIFFRX.bit.RXFIFORESET = 1;
303 } else if (s->got_data) {
304 s->got_data(s, data & 0x00FF);
305 }
306}

370void UART_vprintfLine(unsigned char line, char *format, va_list ap)
371{
372 char sendmsg[24];
373 char buffer[120];
374 int i;
375
376 vsprintf(buffer,format,ap);
377
378 // Add header information and pad end of transfer with spaces to clear display
379 sendmsg[0] = 0xFE;
380 sendmsg[1] = 'G';
381 sendmsg[2] = 1;
382 sendmsg[3] = line;
383 for (i=4;i<24;i++) {
384 if (i >= strlen(buffer)+4) {
385 sendmsg[i] = ' ';
386 } else {
387 sendmsg[i] = buffer[i-4];
388 }
389 }
390 serial_send(&SerialC,sendmsg,24);
391}
